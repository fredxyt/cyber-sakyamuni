SHELL=/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin

# 永动参悟 (cyber-sakyamuni) — 每15分钟一批(略缓, 更合天理); flock 锁在 cron_canpo.sh 内, 串行不重叠。串行不停, 回头, 永不做完。
*/15 * * * * cd /home/ubuntu/cyber-sakyamuni && bash cron_canpo.sh 2 >> logs/cron_canpo_$(date +\%Y\%m\%d).log 2>&1

# 每日札记 — 每天 UTC 23:00 把当天修行策展成今日一篇 (flock 共用同一把锁, 不撞参悟)
0 23 * * * cd /home/ubuntu/cyber-sakyamuni && /usr/bin/flock /tmp/canpo.lock bash cron_daily.sh >> logs/cron_daily_$(date +\%Y\%m\%d).log 2>&1
